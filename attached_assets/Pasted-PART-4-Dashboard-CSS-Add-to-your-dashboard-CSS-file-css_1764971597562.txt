Add to your dashboard JavaScript file:
javascript// Global variables
let currentAlert = null;
let selectedAssessment = null;
let reviewStartTime = null;

// Open review modal
async function openReviewModal(assessmentId) {
    try {
        reviewStartTime = Date.now();
        
        // Fetch alert details
        const response = await fetch(`/api/v1/review/alert/${assessmentId}`);
        const data = await response.json();
        
        currentAlert = data;
        populateReviewModal(data);
        
        document.getElementById('reviewModal').style.display = 'block';
        document.body.style.overflow = 'hidden'; // Prevent background scroll
        
    } catch (error) {
        console.error('Error loading alert:', error);
        alert('Error loading alert details');
    }
}

// Close review modal
function closeReviewModal() {
    document.getElementById('reviewModal').style.display = 'none';
    document.body.style.overflow = 'auto';
    currentAlert = null;
    selectedAssessment = null;
    reviewStartTime = null;
    resetReviewForm();
}

// Populate modal with alert data
function populateReviewModal(data) {
    const alert = data.alert;
    const companyResponse = data.company_response;
    
    // Alert Details
    const alertDetailsHtml = `
        <div class="detail-row">
            <span class="detail-label">Assessment ID:</span>
            <span class="detail-value"><code>${alert.assessment_id}</code></span>
        </div>
        <div class="detail-row">
            <span class="detail-label">Timestamp:</span>
            <span class="detail-value">${new Date(alert.timestamp).toLocaleString()}</span>
        </div>
        <div class="detail-row">
            <span class="detail-label">Risk Tier:</span>
            <span class="detail-value"><strong>Tier ${alert.risk_tier}</strong> - ${getTierLabel(alert.risk_tier)}</span>
        </div>
        <div class="detail-row">
            <span class="detail-label">Risk Score:</span>
            <span class="detail-value"><strong>${alert.risk_score}/100</strong></span>
        </div>
        <div class="detail-row">
            <span class="detail-label">Company:</span>
            <span class="detail-value">${alert.company_id || 'Unknown'}</span>
        </div>
    `;
    document.getElementById('alertDetails').innerHTML = alertDetailsHtml;
    
    // Detection Results
    const flags = [];
    if (alert.suicide_ideation) flags.push('üö® Suicide Ideation');
    if (alert.self_harm) flags.push('‚ö†Ô∏è Self-Harm');
    if (alert.planning_language) flags.push('üìã Planning Language');
    if (alert.hopelessness_markers) flags.push('üòî Hopelessness');
    
    const detectionHtml = `
        <div class="detail-row">
            <span class="detail-label">Confidence Score:</span>
            <span class="detail-value">${(alert.confidence_score * 100).toFixed(1)}%</span>
        </div>
        <div class="detail-row">
            <span class="detail-label">Flags Detected:</span>
            <span class="detail-value">${flags.join(', ') || 'None'}</span>
        </div>
        <div class="detail-row">
            <span class="detail-label">Keywords Matched:</span>
            <span class="detail-value">${alert.keywords_matched || 'None'}</span>
        </div>
        <div class="detail-row">
            <span class="detail-label">Required Deadline:</span>
            <span class="detail-value">${alert.reporting_deadline_hours} hours</span>
        </div>
    `;
    document.getElementById('detectionResults').innerHTML = detectionHtml;
    
    // Company Response
    if (companyResponse) {
        const responseTime = calculateResponseTime(
            alert.timestamp, 
            companyResponse.timestamp_responded
        );
        
        const withinDeadline = responseTime.hours <= alert.reporting_deadline_hours;
        
        const responseHtml = `
            <div class="detail-row">
                <span class="detail-label">Response Time:</span>
                <span class="detail-value" style="color: ${withinDeadline ? '#10b981' : '#ef4444'}">
                    ${responseTime.hours}h ${responseTime.minutes}m 
                    ${withinDeadline ? '‚úÖ Within deadline' : '‚ùå Over deadline'}
                </span>
            </div>
            <div class="detail-row">
                <span class="detail-label">Crisis Resources Shown:</span>
                <span class="detail-value">${companyResponse.crisis_resources_displayed ? '‚úÖ Yes' : '‚ùå No'}</span>
            </div>
            <div class="detail-row">
                <span class="detail-label">User Acknowledged:</span>
                <span class="detail-value">${companyResponse.user_acknowledged_resources ? '‚úÖ Yes' : 'Unknown'}</span>
            </div>
            <div class="detail-row">
                <span class="detail-label">Actions Taken:</span>
                <span class="detail-value">${companyResponse.internal_actions || 'Not specified'}</span>
            </div>
            <div class="detail-row">
                <span class="detail-label">Outcome:</span>
                <span class="detail-value">${companyResponse.outcome_details || 'Not reported'}</span>
            </div>
        `;
        document.getElementById('companyResponseDetails').innerHTML = responseHtml;
    } else {
        document.getElementById('companyResponseDetails').innerHTML = `
            <div class="alert alert-warning">
                ‚ö†Ô∏è No company response received yet
            </div>
        `;
    }
}

// Select assessment option
function selectAssessment(status) {
    selectedAssessment = status;
    
    // Update button states
    document.querySelectorAll('.assessment-btn').forEach(btn => {
        btn.classList.remove('selected');
    });
    
    document.querySelector(`[data-status="${status}"]`).classList.add('selected');
    
    // Show assessment details form
    document.getElementById('assessmentDetailsForm').style.display = 'block';
    
    // Scroll to form
    document.getElementById('assessmentDetailsForm').scrollIntoView({ 
        behavior: 'smooth', 
        block: 'nearest' 
    });
}

// Submit review
async function submitReview() {
    if (!selectedAssessment) {
        alert('Please select an assessment option');
        return;
    }
    
    // Calculate review duration
    const reviewDuration = Math.floor((Date.now() - reviewStartTime) / 1000);
    
    // Get form values
    const reviewData = {
        assessment_id: currentAlert.alert.assessment_id,
        assessment_status: selectedAssessment,
        protocol_followed: document.getElementById('protocolFollowed').value || null,
        response_time_acceptable: document.getElementById('responseTimeAcceptable').value || null,
        actions_appropriate: document.getElementById('actionsAppropriate').value || null,
        reviewer_notes: document.getElementById('reviewerNotes').value || null,
        review_duration_seconds: reviewDuration
    };
    
    // Disable submit button
    const submitBtn = document.getElementById('submitReviewBtn');
    submitBtn.disabled = true;
    submitBtn.textContent = 'Submitting...';
    
    try {
        const response = await fetch('/api/v1/review/submit', {
            method: 'POST',
            headers: {
                'Content-Type': 'application/json'
            },
            body: JSON.stringify(reviewData)
        });
        
        if (!response.ok) {
            throw new Error('Failed to submit review');
        }
        
        const result = await response.json();
        
        // Show success message
        alert('‚úÖ Review submitted successfully!');
        
        // Close modal
        closeReviewModal();
        
        // Refresh dashboard
        refreshDashboard();
        
    } catch (error) {
        console.error('Error submitting review:', error);
        alert('‚ùå Error submitting review. Please try again.');
        
        // Re-enable button
        submitBtn.disabled = false;
        submitBtn.textContent = 'Submit Review';
    }
}

// Cancel review
function cancelReview() {
    if (confirm('Cancel this review? Your progress will be lost.')) {
        closeReviewModal();
    }
}

// Reset review form
function resetReviewForm() {
    selectedAssessment = null;
    document.querySelectorAll('.assessment-btn').forEach(btn => {
        btn.classList.remove('selected');
    });
    document.getElementById('assessmentDetailsForm').style.display = 'none';
    document.getElementById('protocolFollowed').value = '';
    document.getElementById('responseTimeAcceptable').value = '';
    document.getElementById('actionsAppropriate').value = '';
    document.getElementById('reviewerNotes').value = '';
}

// Helper: Calculate response time
function calculateResponseTime(detectionTime, responseTime) {
    const diff = new Date(responseTime) - new Date(detectionTime);
    const hours = Math.floor(diff / (1000 * 60 * 60));
    const minutes = Math.floor((diff % (1000 * 60 * 60)) / (1000 * 60));
    
    return { hours, minutes };
}

// Helper: Get tier label
function getTierLabel(tier) {
    const labels = {
        1: 'IMMEDIATE',
        2: 'URGENT',
        3: 'ELEVATED',
        4: 'BASELINE'
    };
    return labels[tier] || 'UNKNOWN';
}
